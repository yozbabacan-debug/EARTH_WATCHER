let countriesLayer;

map.on('mousemove', function(e) {
  const lat = e.latlng.lat;
  const lng = e.latlng.lng;
  
  // Get country at mouse position
  getCountryAtLocation(lat, lng);
});

function getCountryAtLocation(lat, lng) {
  // Highlight country under cursor
  if (countriesLayer) {
    countriesLayer.eachLayer(layer => {
      if (layer.feature.geometry.type === 'Polygon' || 
          layer.feature.geometry.type === 'MultiPolygon') {
        const isInside = pointInPolygon([lng, lat], layer.feature.geometry);
        if (isInside) {
          layer.setStyle({ fillOpacity: 0.8, weight: 3 });
        } else {
          layer.setStyle({ fillOpacity: 0.3, weight: 2 });
        }
      }
    });
  }
}

Better Solution: Use Turf.js (Geospatial Library)
For easier point-in-polygon detection:

JavaScript

// Add Turf.js library
<script src='https://cdn.jsdelivr.net/npm/@turf/turf@6/turf.min.js'></script>

map.on('mousemove', function(e) {
  const point = turf.point([e.latlng.lng, e.latlng.lat]);
  
  countriesLayer.eachLayer(layer => {
    const isInside = turf.booleanPointInPolygon(point, layer.feature);
    if (isInside) {
      layer.setStyle({ 
        fillOpacity: 0.8, 
        weight: 3,
        color: '#FFD700' // Gold border
      });
    } else {
      layer.setStyle({ 
        fillOpacity: 0.3, 
        weight: 2,
        color: '#fff'
      });
    }
  });
});
